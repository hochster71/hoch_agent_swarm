    function nowISO() {

    function log(msg, type = 'info') {
      const el = document.getElementById('heartbeat-log');
      if (!el) return;
      const color = type === 'ok' ? '#4ade80' : type === 'err' ? '#f87171' : '#22d3ee';
      const entry = document.createElement('div');
      entry.className = 'hb-entry';
      entry.innerHTML = `<span class="hb-ts" style="color:${color}">[${nowISO()}]</span> ${msg}`;
      el.prepend(entry);
      while (el.children.length > 40) el.removeChild(el.lastChild);
    }

    function setGate(status, label, sub) {

      document.getElementById('gate-sub').textContent = sub;

      try {
        const r = await fetch(`${BASE}/health`, { cache: 'no-store' });
        if (!r.ok) throw new Error(`HTTP ${r.status}`);
        const d = await r.json();
        return { ok: true, data: d };
      } catch (e) {
        log(`/health error: ${e.message}`, 'err');
        return { ok: false, message: e.message };
      }
    }
